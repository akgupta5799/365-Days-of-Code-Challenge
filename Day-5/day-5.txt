Question link: https://www.interviewbit.com/problems/rotated-array/

public class Solution {
	// DO NOT MODIFY THE LIST
public int findMin(final List<Integer> a) {
       int minEle = a.get(0);
       int left = 0;
       int right = a.size() - 1;
       while(left <= right){
           int mid = left + (right - left)/2;
           if(a.get(mid) <= minEle){
               minEle = a.get(mid);
               right = mid - 1;
           }else{
               left = mid + 1;
           }
       }
       return minEle;
	}
}
